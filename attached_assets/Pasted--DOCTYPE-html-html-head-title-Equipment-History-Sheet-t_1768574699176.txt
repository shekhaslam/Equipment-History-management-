<script>
  function addRow() {
    const table = document.getElementById('repairTable');
    const row = table.insertRow();

    row.insertCell(0).innerHTML = '<input type="date">';
    row.insertCell(1).innerHTML = '<input type="text" placeholder="Cartridge refill / servicing">';
    row.insertCell(2).innerHTML = '<input type="number">';
  }

  function generatePDF() {
    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF();

    let y = 15;
    pdf.setFontSize(14);
    pdf.text('Equipment History Sheet', 70, y);

    y += 15;
    pdf.setFontSize(11);
    pdf.text('Office Name: ' + officeName.value, 20, y); y+=8;
    pdf.text('Division: ' + division.value, 20, y); y+=8;
    pdf.text('Area: ' + area.value, 20, y); y+=8;
    pdf.text('Pincode: ' + pincode.value, 20, y); y+=12;

    pdf.text('Equipment Name: ' + equipmentName.value, 20, y); y+=8;
    pdf.text('Model Number: ' + modelNumber.value, 20, y); y+=8;
    pdf.text('Serial Number: ' + serialNumber.value, 20, y); y+=8;
    pdf.text('Monthly Usage / Refill Frequency: ' + usage.value, 20, y); y+=12;

    pdf.text('Repair and Maintenance History:', 20, y); y+=8;

    const rows = document.querySelectorAll('#repairTable tr');
    for (let i = 1; i < rows.length; i++) {
      const cells = rows[i].querySelectorAll('input');
      pdf.text(cells[0].value + ' | ' + cells[1].value + ' | INR ' + cells[2].value, 20, y);
      y += 7;
    }

    y += 10;
    pdf.text('Remarks:', 20, y); y+=8;
    pdf.text('The above equipment is maintained regularly and repair details are recorded for official and audit purposes.', 20, y, { maxWidth:170 });

    y += 20;
    pdf.text('Signature of Incharge', 20, y);
    pdf.text('Date:', 140, y);

    pdf.save('Equipment_History_Sheet.pdf');
  }
</script>
